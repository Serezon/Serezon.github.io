Оператори порівняння

Більше менше


  const x = 7;

  console.log(x > 5); // -> true
  console.log(x < 0); // -> false

  console.log("b" > "a"); // -> true
  console.log("B" > "a"); // -> false

Більше менше або рівне


  const x = 7;

  console.log(x >= 7); // -> true
  console.log(x <= 7); // -> true

Рівність (строга і нестрога)


  console.log(2 == 2); // -> true

  console.log(2 == "2"); // -> true
  console.log(1 == true); // -> true
  console.log(0 == ""); // -> true

  console.log(2 === "2"); // -> false

Нерівність (строга і нестрога)


  const x = 2020;

  console.log(x != "2020"); // -> false
  console.log(x !== "2020"); // -> true

Методи роботи з масивами

pop/push, shift/unshift


  const arr = [1,1,2,3,5,8,13];

  console.log(arr.pop()); // 13
  console.log(arr); // [1,1,2,3,5,8]

  console.log(arr.push(13, 13)); // 8
  console.log(arr); // [1,1,2,3,5,8,13,13]

  console.log(arr.shift()); // 1
  console.log(arr); // [1,2,3,5,8,12,13]

  console.log(arr.unshift(0, 1)); // 9
  console.log(arr); // [0, 11,2,3,5,8,12,13]

slice


  const arr = [1,1,2,3,5,8,13];

  console.log(arr.slice()); // [1,1,2,3,5,8,13]
  console.log(arr.slice(0, 3)); // [1,2,3]
  console.log(arr.slice(-3)); // [5,8,13]

splice


  const arr = [1,2,3,5,7,11,13,17];

  console.log(arr.splice(0, 2)); // [1,2]
  console.log(arr); // [3,5,7,11,13,17]

  console.log(arr.splice(2, 0, 22, 33)); // []
  console.log(arr); // [3,5,22,33,7,11,13,17]

concat


  const arr = [1,3,5,7,9];

  console.log(arr.concat([2,4,6,8])); // [1,3,5,7,9,2,4,6,8]

find, findIndex


  const arr = [1,2,4,8,16,32,64];

  console.log(arr.find(x => x > 10)); // 16
  console.log(arr.find(x => x > 100)); // undefined

  console.log(arr.findIndex(x => x > 10)); // 4
  console.log(arr.findIndex(x => x > 100)); // -1

indexOf, lastIndexOf, includes


  const arr = ["Bob", "John", "Joe", "Dan", "John"];

  console.log(arr.indexOf("John")); // 1
  console.log(arr.indexOf("Lenny")); // -1

  console.log(arr.lastIndexOf("John")); // 4

  console.log(arr.includes("Joe")); // true

some, every


  const arr = [-1,2,-3,4,5];

  console.log(arr.some(
    x => x > 0
  )); // true

  console.log(arr.every(
    x => x > 0
  )); // false

sort, reverse


  const arr = ["Bob", "John", "Joe", "Dan"];

  console.log(arr.reverse()); // ['Dan','Joe','John','Bob']
  console.log(arr); // ['Dan','Joe','John','Bob']

  console.log(arr.sort()); // ['Bob','Dan','Joe','John']
  console.log(arr); // ['Bob','Dan','Joe','John']

  const names = ["Po", "John", "Dan", "Hulio"];
  console.log(names.sort(
    (a,b) => a.length > b.length ? 1 : -1)
  );
  // ['Po','Dan','John','Hulio']
